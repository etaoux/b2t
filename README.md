```bash
b2t init # 初始化 channel.json

# 在package目录中

b2t page-add home # 添加home页面
b2t page-remove home
b2t page-list # 当前package所有页面

b2t component-add header # 初始化component.js
b2t component-remove header 
b2t component-list # 当前package所有组件

# 在component目录中
b2t skin-add default
b2t skin-remove default
b2t skin-list # 某组件所有皮肤
```
